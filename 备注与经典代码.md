### 1、输入string，得到int等整数

~~~C++
int getInt() {
	char ch = getchar();
	int x = 0, f = 1;
	while (ch < '0' || ch>'9') {
		if (ch == '-') f = -1;
		ch = getchar();
	}
	while (ch >= '0' && ch <= '9') {
		x = x * 10 + ch - '0';
		ch = getchar();
	}

	return x * f;
}
~~~
